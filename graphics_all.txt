//Mid Point Algorithm- Circle

#include <iostream>
#include <windows.h>
#include <GL/gl.h>
#include <GL/glut.h>
using namespace std;

int center_x, center_y, r;

void plotting_points(int x, int y)
{
    glBegin(GL_POINTS);
    glVertex2i(center_x + x, center_y + y);
    glVertex2i(center_x + y, center_y + x);
    glVertex2i(center_x + y, center_y - x);
    glVertex2i(center_x + x, center_y - y);
    glVertex2i(center_x - x, center_y - y);
    glVertex2i(center_x - y, center_y - x);
    glVertex2i(center_x - y, center_y + x);
    glVertex2i(center_x - x, center_y + y);
    glEnd();
}

void circle(int x, int y, int r)
{
    int xk = 0;
    int yk = r;
    int pk = 1 - r;

    plotting_points(xk, yk);

    while (yk >= xk)
    {


        if (pk < 0)
        {
            pk = pk + (2 * xk) + 3;
            xk = xk + 1;
            yk=yk;
        }
        else
        {

            pk = pk + (2 * xk) - (2 * yk) + 5;
            xk = xk + 1;
            yk=yk-1;
        }

        plotting_points(xk, yk);
    }
}

void init()
{
    glClearColor(0.0, 0.0, 0.0, 1.0);
    glMatrixMode(GL_PROJECTION);
    gluOrtho2D(-10.0, 10.0, -10.0, 10.0);
}

void display(void) {
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3f(1.0, 0.0, 0.0);
    glPointSize(3.0);

    circle(center_x, center_y, r);

    glFlush();
}

int main(int argc, char** argv)
{
    center_x = 0;
    center_y = 0;
    r = 5;

    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutInitWindowPosition(100, 100);
    glutCreateWindow("Circle Drawing using Bresenham's Algorithm");

    init();
    glutDisplayFunc(display);
    glutMainLoop();

    return 0;
}

//Breesenham Algorithm

#include <iostream>
#include <windows.h>
#include <GL/gl.h>
#include <GL/glut.h>
using namespace std;

int x_start, y_start, x_end, y_end;

void init()
{
    glClearColor(0.0, 0.0, 0.0, 1.0);
    glMatrixMode(GL_PROJECTION);
    gluOrtho2D(0.0, 300.0, 0.0, 300.0);
}

void display(void) {
    glClear(GL_COLOR_BUFFER_BIT);

    glColor3f(1.0, 0.302, 0.004);
    glPointSize(3.0);

    int dx = abs(x_end - x_start);
    int dy = abs(y_end - y_start);

    int xk = x_start;
    int yk = y_start;

    int pk = 2 * dy - dx;


    glBegin(GL_POINTS);
    glVertex2i(xk, yk);

    for (int k = 0; k < dx; k++)
    {

        if (pk < 0)
        {
            pk = pk+ (2 * dy);
        }
        else
        {
            yk = yk + 1;
            pk = pk+ (2 *dy) - (2*dx);
        }

        xk=xk+1;

        glVertex2i(xk, yk);
    }

    glEnd();
    glFlush();
}

int main(int argc, char** argv)
{
    x_start = 50;
    y_start = 150;
    x_end = 180;
    y_end = 280;

    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutInitWindowPosition(100, 100);
    glutCreateWindow("Line Drawing using Bresenham's Algorithm");

    init();
    glutDisplayFunc(display);
    glutMainLoop();

    return 0;
}

//DDA LINE Drawing 
#include<iostream>
#include<windows.h>
#include<GL/gl.h>
#include <GL/glut.h>
using namespace std;

float x_start,y_start,x_end,y_end,x_inc,y_inc,x_curr,y_curr;
float dx,dy,steps,m,i;

void init()
{
    glClearColor(0.0,0.0,0.0,1.0);
    glMatrixMode(GL_PROJECTION);
    gluOrtho2D(0.0,100.0,0.0,100.0);

}

void display(void)
{
    glClear(GL_COLOR_BUFFER_BIT);

    glColor3f(1.0,0.302,0.004);
    glPointSize(3.0);

    glBegin(GL_POINTS);

    dx=x_end-x_start;
    dy=y_end-y_start;

    m=dy/dx;

    if(abs(dx)>=abs(dy))
    {
        steps=dx;

    }
    else
    {
        steps=dy;

    }

    if(0<m && m<=1)
    {
        x_inc=1;
        y_inc=m;
    }
    else if(-1<=m && m<=0)
    {
        x_inc=-1;
        y_inc=-m;

    }
    else if(m>1)
    {
        x_inc=1/m;
        y_inc=1;

    }
    else if(m<-1)
    {
        cout<<"Error";

    }

    for(i=1; i<=steps; i++)
    {
        glVertex2i(x_curr,y_curr);
        cout<<i<<" "<<x_curr<<" "<<" "<<y_curr<<endl;

        x_curr=x_curr+x_inc;
        y_curr=y_curr+y_inc;

    }

    glEnd();
    glFlush();



}
int main(int argc, char** argv)
{
    x_start=20;
    y_start=20;
    x_end=28;
    y_end=24;

    glutInit(&argc,argv);
    glutInitDisplayMode(GLUT_SINGLE|GLUT_RGB);
    glutInitWindowSize(500,500);
    glutInitWindowPosition(100,100);
    glutCreateWindow("Line drawing in Glut with slope m:");

    init();
    glutDisplayFunc(display);
    glutMainLoop();

    return 0;


}
